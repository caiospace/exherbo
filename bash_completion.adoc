== Example

----
_mut()
{
    local cur prev opts
    COMPREPLY=()
    cur="${COMP_WORDS[COMP_CWORD]}"
    prev="${COMP_WORDS[COMP_CWORD-1]}"
    opts="--nologfile --offline --help"

    if [[ ${cur} == -* ]] ; then
        COMPREPLY=( $(compgen -S " " -W "${opts}" -- ${cur}) )
        return 0
    fi
}
complete -o default -F _mut mut
----
